@GeneratedValue

кё°ліё нӮӨ(primary key) к°’мқҳ мғқм„ұ м „лһөмқ„ м§Җм •н•ҳлҠ” лҚ° мӮ¬мҡ©
мҰү, лҚ°мқҙн„°лІ мқҙмҠӨм—җ мғҲ л ҲмҪ”л“ңлҘј мӮҪмһ…н•  л•Ң кё°ліё нӮӨ к°’мқ„ мһҗлҸҷмңјлЎң мғқм„ұн•ҳлҠ” л°©лІ•мқ„ м§Җм •

  • @GeneratedValueлҠ” @IdмҷҖ н•Ёк»ҳ м—”нӢ°нӢ° лҳҗлҠ” л§Өн•‘лҗң мҠҲнҚјнҒҙлһҳмҠӨмқҳ кё°ліё нӮӨ мҶҚм„ұмқҙлӮҳ н•„л“ңм—җ м Ғмҡ©н•  мҲҳ мһҲлӢӨ.
  • @GeneratedValueлҠ” лӢЁмҲң кё°ліё нӮӨ(лӢЁмқј м»¬лҹјмңјлЎң кө¬м„ұлҗң PK)м—җ лҢҖн•ҙм„ңл§Ң м§ҖмӣҗлҗңлӢӨ.
    • нҢҢмғқ кё°ліё нӮӨ(@MapsIdлЎң лӢӨлҘё м—”нӢ°нӢ°лЎңл¶Җн„° нҢҢмғқлҗң PK)м—җ лҢҖн•ң @GeneratedValue мӮ¬мҡ©мқҖ м§Җмӣҗлҗҳм§Җ м•ҠлҠ”лӢӨ.

// org.springframework.data.jpa.repository.support.SimpleJpaRepository
@Transactional
@Override
public <S extends T> S save(S entity) {

    Assert.notNull(entity, "Entity must not be null");

    if (entityInformation.isNew(entity)) {
        // мғҲ м—”нӢ°нӢ°л©ҙ persist
        entityManager.persist(entity);
        return entity;
    } else {
        // кё°мЎҙ м—”нӢ°нӢ°л©ҙ merge
        return entityManager.merge(entity);
    }
}

public void onPersist(PersistEvent event, PersistContext createdAlready) {
final Object object = event.getObject();

    // м—”нӢ°нӢ° мғҒнғң нҷ•мқё
    final EntityEntry entityEntry = event.getSession()
        .getPersistenceContextInternal()

// org.hibernate.event.internal.AbstractSaveEventListener
protected Object saveWithGeneratedId(Object entity, String entityName, Object anything,
EventSource source, boolean requiresImmediateIdAccess) {

    // м—”нӢ°нӢ° л©”нғҖлҚ°мқҙн„° к°Җм ёмҳӨкё°
    EntityPersister persister = source.getEntityPersister(entityName, entity);

    // вҳ… ID мғқм„ұкё° к°Җм ёмҷҖм„ң ID мғқм„ұ
    Object generatedId = persister.getIdentifierGenerator()
        .generate((SharedSessionContractImplementor) source, entity);

    // IDENTITY м „лһөмқё кІҪмҡ° (POST_INSERT_INDICATOR л°ҳнҷҳлҗЁ)
    if (generatedId == IdentifierGeneratorHelper.POST_INSERT_INDICATOR) {
        // INSERT нӣ„ IDлҘј к°Җм ёмҷҖм•ј н•Ё
        return performSaveWithId(entity, null, persister, ...);
    }

    // SEQUENCE, TABLE м „лһөмқё кІҪмҡ° (мқҙлҜё IDк°Җ мғқм„ұлҗЁ)
    return performSaveWithId(entity, generatedId, persister, ...);
}

// org.hibernate.id.insert.AbstractReturningDelegate
public Object performInsert(PreparedStatementDetails insertStatementDetails,
JdbcValueBindings jdbcValueBindings, Object entity, SharedSessionContractImplementor session) {

    // INSERT мӢӨн–ү
    session.getJdbcCoordinator()
        .getResultSetReturn()
        .executeUpdate(insertStatement);

    // DBм—җм„ң мғқм„ұлҗң ID мЎ°нҡҢ
    // MySQL: SELECT LAST_INSERT_ID()
    // PostgreSQL: RETURNING id
    ResultSet rs = getInsertGeneratedIdentifierDelegate()
        .executeAndExtract(insertStatement, session);

    return rs.getLong(1);  // мғқм„ұлҗң ID л°ҳнҷҳ
}
